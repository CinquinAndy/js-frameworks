        {Array.from({ length: count }, (_, i) => (

import { useSpring, animated, useTrail } from '@react-spring/web'
import { useState } from 'react'

function SpringAnimations() {
  const [flip, setFlip] = useState(false)
  const [items, setItems] = useState(['Item 1', 'Item 2', 'Item 3'])

  // Animation simple
  const styles = useSpring({
    transform: flip ? 'scale(1.2) rotate(45deg)' : 'scale(1) rotate(0deg)',
    backgroundColor: flip ? '#ff6b6b' : '#4ecdc4',
  })

  // Animation de trail (effet cascade)
  const trail = useTrail(items.length, {
    from: { opacity: 0, transform: 'translate3d(0,-40px,0)' },
    to: { opacity: 1, transform: 'translate3d(0,0px,0)' },
  })

  return (
    <div className="p-8">
      <h2 className="text-2xl font-bold mb-6">React Spring Animations</h2>

      {/* Box anim√©e */}
      <animated.div
        style={styles}
        onClick={() => setFlip(!flip)}
        className="w-20 h-20 cursor-pointer mb-8 rounded-lg"
      />

      {/* Liste avec effet cascade */}
      <div>
        <button
          onClick={() => setItems([...items, `Item ${items.length + 1}`])}
          className="bg-blue-500 text-white px-4 py-2 rounded mb-4"
        >
          Ajouter Item
        </button>

        {trail.map((style, index) => (
          <animated.div
            key={index}
            style={style}
            className="p-4 bg-gray-100 rounded mb-2"
          >
            {items[index]}
          </animated.div>
        ))}
      </div>
    </div>
  )
}
